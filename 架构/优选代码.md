# 优选代码
```javascript

import { uniqBy } from 'lodash-es'

/**
 *  每个图表的基础配置
 *  import.meta.glob: Vite 提供的功能，用于批量导入匹配的文件
 *  './*.ts': 通配符模式，匹配当前目录下所有 .ts 文件（不包括子目录）0
 *  { eager: true }: 立即同步导入所有模块，而不是懒加载
 */
const modules:any = import.meta.glob('./*.ts', { eager: true })

/** 每个图表的基础配置 */
let modulesConfigs:any[] = []
for (const path in modules) {
 // 通过 fileTemplateCode 字段去重 避免重复配置
  modulesConfigs = uniqBy(modulesConfigs.concat(modules[path].config), 'fileTemplateCode')
}

/** 每个数据项的基本配置 */
export const configList:any[] = (() => {
  return modulesConfigs
})()

```
